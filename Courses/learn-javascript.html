01 كيف تشغل JavaScript؟

JavaScript تعمل مباشرة في أي متصفح. أسهل طريقة هي فتح Developer Tools (F12) والذهاب لتبويب Console وكتابة الكود هناك. أو يمكنك ربطها بملف HTML:

index.html 
<!-- طريقة 1: كود مباشر في HTML -->
<script>
  console.log("مرحبا بالعالم!");
</script>

<!-- طريقة 2: ملف JS خارجي (الأفضل) -->
<script src="app.js"></script>

<!-- دائماً ضع الـ script قبل إغلاق body -->

💡 لماذا نضع script قبل </body>؟

  • المتصفح يقرأ HTML من فوق لتحت — إذا وضعت الـ script في <head> ستحاول الوصول لعناصر HTML لم تُحمَّل بعد
  • الحل: ضعها آخر الـ body، أو استخدم defer أو async في أحدث كود

03 — المتغيرات

Variables — تخزين البيانات

المتغير هو مكان في الذاكرة نخزن فيه بيانات ونستخدمها لاحقاً

04let, const, var — الفرق بينهم

في JavaScript الحديثة (ES6+) لدينا 3 طرق لتعريف المتغيرات. القاعدة الذهبية: استخدم const دائماً، let عند الحاجة للتغيير، وتجنب var.

الكلمة إعادة التعيين إعادة التعريف النطاق (Scope) متى تستخدم؟
const ✗ لا ✗ لا Block القيم الثابتة (الأكثر استخداماً)
let ✓ نعم ✗ لا Block عند الحاجة لتغيير القيمة
var ✓ نعم ✓ نعم Function ❌ تجنبه في الكود الحديث
JavaScript 
// ─── const ───
const siteName = "Yahya.DEV";
// siteName = "غير ذلك"; // ❌ خطأ! لا يمكن تغيير const

// ─── let ───
let score = 0;
score = 100; // ✅ يمكن تغييره
score += 50;  // ✅ score = 150

// ─── تعريف متعدد ───
let x = 10, y = 20, z = 30;

// ─── const مع Objects/Arrays ───
// const لا يمنع تعديل محتوى الـ Object!
const user = { name: "يحيى" };
user.name = "محمد"; // ✅ يمكن تعديل الخصائص
// user = {}; // ❌ لا يمكن إعادة تعيين المتغير نفسه

🧠 ما هو Block Scope؟

  • الـ Block هو أي كود بين { } — داخل if، loop، function
  • المتغير المعرَّف بـ let/const داخل block لا يمكن الوصول إليه من خارجه
  • var يتجاهل الـ block — وهذا يسبب مشاكل كثيرة ولهذا نتجنبه

Functions — قلب JavaScript

14كل أنواع الدوال
JavaScript 
// ─── 1. Function Declaration ───
function greet(name) {
  return `مرحباً ${name}!`;
}
greet("يحيى"); // يمكن استدعاؤها قبل تعريفها (Hoisting)

createUser("يحيى", "admin"); // {name: "يحيى", role: "admin"}

// ─── 5. Rest Parameters ───
const sum = (...nums) => nums.reduce((a, b) => a + b, 0);
sum(1, 2, 3, 4, 5); // 15

// ─── 6. IIFE (Immediately Invoked) ───
(function() {
  console.log("تنفذ فوراً!");
})();
15Scope & Closure — المفهوم الأعمق
JavaScript 
// ─── Closure ── دالة تتذكر بيئتها ───
function createCounter() {
  let count = 0; // متغير خاص — محمي من الخارج
  return {
    increment: () => ++count,
    decrement: () => --count,
    getCount: () => count
  };
}

const counter = createCounter();
counter.increment(); // 1
counter.increment(); // 2
counter.getCount(); // 2
// لا يمكن الوصول لـ count من الخارج!

🔐 لماذا Closure مهم؟

  • يحمي البيانات ويجعلها خاصة (Encapsulation)
  • يُستخدم في بناء مكتبات مثل jQuery
  • أساسي في فهم React Hooks

16Array Methods الأساسية والمتقدمة
JavaScript 
const nums = [1, 2, 3, 4, 5];

// ─── إضافة وحذف ───
nums.push(6);         // إضافة للنهاية   → [1,2,3,4,5,6]
nums.pop();           // حذف من النهاية  → [1,2,3,4,5]
nums.unshift(0);      // إضافة للبداية   → [0,1,2,3,4,5]
nums.shift();         // حذف من البداية  → [1,2,3,4,5]
nums.splice(1, 2);    // حذف 2 عناصر من index 1

// ─── البحث ───
nums.includes(3);     // true
nums.indexOf(3);      // 2
nums.find(n => n > 3);     // 4 (أول عنصر يطابق)
nums.findIndex(n => n > 3); // 3 (index أول عنصر)

// ─── التحويل (الأهم!) ───
nums.map(n => n * 2);   // [2,4,6,8,10] — تحويل كل عنصر
nums.filter(n => n > 2); // [3,4,5] — تصفية
nums.reduce((acc, n) => acc + n, 0); // 15 — تجميع
nums.forEach(n => console.log(n));  // لا يرجع قيمة
nums.some(n => n > 4);   // true — هل أحد العناصر؟
nums.every(n => n > 0);  // true — هل كلهم؟
nums.sort((a, b) => a - b);  // ترتيب تصاعدي
nums.flat();                   // تسطيح Array متداخل
nums.flatMap(n => [n, n * 2]); // map + flat معاً

// ─── نسخ ودمج ───
const copy = [...nums];          // نسخ بالـ Spread
const merged = [...nums, ...[6,7]]; // دمج مصفوفتين
Array.from({ length: 5 }, (_, i) => i); // [0,1,2,3,4]

22Async/Await — الأسلوب الحديث
JavaScript 
// Fetch API + Async/Await
async function getUsers() {
  try {
    const response = await fetch("https://api.example.com/users");

    if (!response.ok) {
      throw new Error(`HTTP Error: ${response.status}`);
    }

    const users = await response.json();
    console.log(users);
    return users;

  } catch (error) {
    console.error("خطأ:", error.message);
  } finally {
    console.log("انتهى الطلب");
  }
}

// Arrow async function
const postData = async (url, data) => {
  const res = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(data)
  });
  return res.json();
};

23Classes — ES6+
JavaScript 
class Developer {
  // Private fields (ES2022)
  #salary = 0;

  constructor(name, role) {
    this.name = name;
    this.role = role;
  }

  // Getter
  get info() {
    return `${this.name} — ${this.role}`;
  }

  // Method
  greet() {
    return `مرحباً! أنا ${this.name}`;
  }

  // Static method
  static create(name) {
    return new Developer(name, "Junior");
  }
}

// Inheritance — الوراثة
class FullStack extends Developer {
  constructor(name) {
    super(name, "Full-Stack"); // استدعاء الـ parent
    this.stack = ["Laravel", "React", "MySQL"];
  }

  greet() {
    return `${super.greet()} — Full-Stack Developer!`;
  }
}

const yahya = new FullStack("يحيى");
yahya.greet();        // "مرحباً! أنا يحيى — Full-Stack Developer!"
yahya instanceof Developer; // true

Error Handling

JavaScript
class ApiError extends Error {
  constructor(msg, status) {
    super(msg);
    this.status = status;
  }
}
try {
  throw new ApiError("Unauthorized", 401);
} catch (e) {
  if (e instanceof ApiError) {
    console.log(e.status); // 401
  }
}
